          Text(
            '${tab.title}ÂÜÖÂÆπ',
            style: TextStyle(
              fontSize: 24,
              fontWeight: FontWeight.bold,
            ),

          SizedBox(height: 16),

            ElevatedButton(
              onPressed: () => _removeTab(tab),
              child: Text('ÂÖ≥Èó≠Ê≠§Ê†áÁ≠æÈ°µ'),
              style: ElevatedButton.styleFrom(
                backgroundColor: Colors.red,
                foregroundColor: Colors.white,
              ),
            ),
        ],
      ),
    );
  }

  void _addTab() {
    final newTabIndex = _tabs.length + 1;
    final newTab = TabData('È°µÈù¢ $newTabIndex', Icons.tab);
    
    setState(() {
      _tabs.add(newTab);
      _tabController = TabController(length: _tabs.length, vsync: this);
    });
    
    // ÂàáÊç¢Âà∞Êñ∞Ê∑ªÂä†ÁöÑÊ†áÁ≠æÈ°µ
    _tabController.animateTo(_tabs.length - 1);
  }

  void _removeTab(TabData tab) {
    if (_tabs.length <= 1) return;
    
    final currentIndex = _tabController.index;
    final removeIndex = _tabs.indexOf(tab);
    
    setState(() {
      _tabs.remove(tab);
      _tabController = TabController(length: _tabs.length, vsync: this);
    });
    
    // Â¶ÇÊûúÂà†Èô§ÁöÑÊòØÂΩìÂâçÊ†áÁ≠æÈ°µÔºåÂàáÊç¢Âà∞Ââç‰∏Ä‰∏™Ê†áÁ≠æÈ°µ
    if (removeIndex == currentIndex && currentIndex > 0) {
      _tabController.animateTo(currentIndex - 1);
    } else if (removeIndex < currentIndex) {
      _tabController.animateTo(currentIndex - 1);
    }
  }
